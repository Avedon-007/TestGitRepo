abstract class Cinema extends Entertainments {
			private String moviegoer;		
			private String movie;
			private String screen;
	
	public String getMoviegoer() {
		return moviegoer;
		}
	public void setMoviegoer(String newMoviegoer) {
		moviegoer=newMoviegoer;
		}		
			
	public String getMovie() {
		return movie;
		}
	public void setMovie(String newMovie) {
		movie=newMovie;
		}		
			
	public String getScreen() {
		return screen;
		}
	public void setScreen(String newScreen) {
		screen=newScreen;
		}		
		
	public void wathc() {}
	public void sit() {}
	
}
